Stop Configuring.

DockRouter discovers your containers, provisions TLS certificates, and routes traffic — automatically. No config files, just labels.

Get Started → View on GitHub
docker-compose.yml 
# That's it. HTTPS is live.
services:
  dockrouter:
    image: ghcr.io/dockrouter/dockrouter:latest
    ports: ["80:80", "443:443"]
    volumes: ["/var/run/docker.sock:/var/run/docker.sock:ro"]
    environment:
      DR_ACME_EMAIL: you@example.com

  api:
    image: myapp/api
    labels:
      dr.enable: "true"
      dr.host: "api.example.com"
      dr.tls: "auto"
